                                <p>The Corporate Sustainability Reporting Directive (CSRD) is a new EU regulation that significantly expands sustainability reporting requirements for companies. It replaces the Non-Financial Reporting Directive (NFRD) and introduces more detailed reporting requirements on environmental, social, and governance (ESG) matters.</p>
                                
                                <h3 style="margin: 20px 0 10px; color: var(--accent);">Why CSRD is Mandatory</h3>
                                <p>The CSRD is part of the EU's broader sustainable finance agenda and the European Green Deal. It aims to:</p>
                                <ul style="margin: 15px 0 15px 20px;">
                                    <li>Increase transparency on companies' sustainability performance</li>
                                    <li>Provide investors with comparable sustainability information</li>
                                    <li>Redirect capital flows toward sustainable investments</li>
                                    <li>Support the EU's climate and environmental objectives</li>
                                </ul>
                                
                                <h3 style="margin: 20px 0 10px; color: var(--accent);">Key Requirements</h3>
                                <p>Under CSRD, companies must report according to European Sustainability Reporting Standards (ESRS), covering:</p>
                                <ul style="margin: 15px 0 15px 20px;">
                                    <li>Environmental factors (climate change, pollution, biodiversity)</li>
                                    <li>Social factors (working conditions, equality, human rights)</li>
                                    <li>Governance factors (business ethics, corporate culture)</li>
                                    <li>Double materiality assessment (impact of sustainability issues on the company and the company's impact on society and environment)</li>
                                </ul>
                                
                                <p>Non-compliance can result in significant penalties, reputational damage, and exclusion from certain markets. The directive applies not only to EU companies but also to non-EU companies with substantial activities in the EU market.</p>

        // Image slider functionality
        let slideIndex = 0;
        let slideInterval;

        function startSlider() {
            const slides = document.querySelectorAll('.slide');
            const indicators = document.querySelectorAll('.indicator');
            
            // Hide all slides
            slides.forEach(slide => {
                slide.classList.remove('active');
            });
            
            // Remove active class from all indicators
            indicators.forEach(indicator => {
                indicator.classList.remove('active');
            });
            
            // Show current slide
            slides[slideIndex].classList.add('active');
            indicators[slideIndex].classList.add('active');
            
            // Move to next slide
            slideIndex = (slideIndex + 1) % slides.length;
        }

        function currentSlide(n) {
            clearInterval(slideInterval);
            slideIndex = n;
            startSlider();
            slideInterval = setInterval(startSlider, 5000);
        }

        // Show specific section
        function showSection(sectionId) {
            // Hide all sections
            document.querySelectorAll('.content-section').forEach(section => {
                section.classList.remove('active');
            });
            
            // Remove active class from all buttons
            document.querySelectorAll('.nav-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            
            // Show the requested section
            document.getElementById(sectionId).classList.add('active');
            
            // Add active class to clicked button (if it exists)
            if (event && event.target) {
                // Find corresponding top nav button
                const navButtons = document.querySelectorAll('.nav-btn');
                navButtons.forEach(btn => {
                    if (btn.getAttribute('onclick') === `showSection('${sectionId}')`) {
                        btn.classList.add('active');
                    }
                });
            }
            
            // Scroll to top
            window.scrollTo(0, 0);
            
            // Initialize chart if showing dashboard
            if (sectionId === 'dashboard') {
                initializeChart();
            }
            
            // Start slider if showing home section
            if (sectionId === 'home') {
                clearInterval(slideInterval);
                slideInterval = setInterval(startSlider, 5000);
            } else {
                clearInterval(slideInterval);
            }
        }

        // Toggle service details
        function toggleServiceDetails(card) {
            // Close all other service cards
            document.querySelectorAll('.service-card').forEach(serviceCard => {
                if (serviceCard !== card) {
                    serviceCard.classList.remove('active');
                }
            });
            
            // Toggle the clicked card
            card.classList.toggle('active');
        }

        // Calculate compliance
        function calculateCompliance() {
            // Get input values
            const employees = parseInt(document.getElementById('employees').value) || 0;
            const revenue = parseFloat(document.getElementById('revenue').value) || 0;
            const assets = parseFloat(document.getElementById('assets').value) || 0;
            
            // Check criteria (meeting 2 out of 3 triggers compliance)
            const meetsEmployeeCriteria = employees >= 250;
            const meetsRevenueCriteria = revenue >= 40;
            const meetsAssetsCriteria = assets >= 20;
            
            const criteriaMet = [meetsEmployeeCriteria, meetsRevenueCriteria, meetsAssetsCriteria].filter(c => c).length;
            const mustComply = criteriaMet >= 2;
            
            // Get result elements
            const resultIcon = document.getElementById('result-icon');
            const resultTitle = document.getElementById('result-title');
            const resultDescription = document.getElementById('result-description');
            
            // Update results based on compliance check
            if (mustComply) {
                resultIcon.className = 'fas fa-exclamation-triangle result-icon';
                resultTitle.textContent = 'CSRD Compliance Required';
                resultTitle.style.color = '#FF6B6B';
                resultDescription.innerHTML = `Your business <strong>must comply</strong> with EU CSRD reporting requirements as it meets ${criteriaMet} out of 3 criteria.`;
            } else {
                resultIcon.className = 'fas fa-check-circle result-icon';
                resultTitle.textContent = 'CSRD Compliance Not Required';
                resultTitle.style.color = '#4CAF50';
                resultDescription.innerHTML = `Your business does not currently need to comply with CSRD as it meets only ${criteriaMet} out of 3 required criteria.`;
            }
        }

            // Add scroll animations
            const observerOptions = {
                threshold: 0.1,
                rootMargin: '0px 0px -50px 0px'
            };
            
            const observer = new IntersectionObserver(function(entries) {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        entry.target.style.animation = 'fadeInUp 0.8s ease forwards';
                        observer.unobserve(entry.target);
                    }
                });
            }, observerOptions);
            
            // Observe all section cards
            document.querySelectorAll('.section-card').forEach(card => {
                observer.observe(card);
            });
        });
